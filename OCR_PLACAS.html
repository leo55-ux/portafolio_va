import cv2
import pytesseract
#"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"
#C:\Program Files\Tesseract-OCR\tesseract.exe
pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

#lista de palcas aceptadas en el estacionamiento del cbtis 144
placas_autorizadas=["XYZ789","ABC123","FGH589","ABC891","ZYX678"]

#cargar placa a leer
imagenes=["placa1.png","placa2.png"]
placas_analizadas=[]

for imagen in imagenes:
    placa=cv2.imread(imagen)
    placa_nueva=cv2.resize(placa,(800,300))

#separamos el fondo del texto
    placa_gris=cv2.cvtColor(placa_nueva,cv2.COLOR_BGR2GRAY)

#separamos el fondo
    placa_res=cv2.threshold(placa_gris,120,255,cv2.THRESH_BINARY)[1]

#configuramos OCR
    texto_placa=pytesseract.image_to_string(placa_res,lang="spa")

#limpiar la cadena de texto
    texto_limpio=texto_placa.strip().replace(" ","")

#buscar la placa en la lista autoriza
    if texto_limpio in placas_autorizadas:
        placas_analizadas.append(texto_limpio)
    else:
        print("ACCESO DENEGADO")
    cv2.imshow(imagen, placa)

n=1
print("PLACAS ANALIZADAS",placas_analizadas)
print("PLACAS FALTANTES")
for p in placas_autorizadas:
    if not(p in placas_analizadas):
        print(n,".",p)
        n=n+1

cv2.waitKey(0)
cv2.destroyAllWindows()
